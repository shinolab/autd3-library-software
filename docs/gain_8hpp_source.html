30 inline void SignalDesign(uint8_t amp_i, uint8_t phase_i, uint8_t *const amp_o, uint8_t *const phase_o) noexcept {
31  auto d = asin(amp_i / 255.0) / M_PI; // duty (0 ~ 0.5)
32  *amp_o = static_cast<uint8_t>(511 * d);
33  *phase_o = static_cast<uint8_t>(static_cast<int>(phase_i + 64 - 128 * d) % 256);
34 }
