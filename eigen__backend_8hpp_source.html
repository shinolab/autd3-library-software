91  template <typename T, typename E>
92  static std::shared_ptr<T> allocate_matrix_impl(const std::string& name, const size_t row, const size_t col,
93  std::unordered_map<std::string, std::shared_ptr<T>>& cache) {
94  if (const auto it = cache.find(name); it != cache.end()) {
95  if (static_cast<size_t>(it->second->data.rows()) == row && static_cast<size_t>(it->second->data.cols()) == col) return it->second;
96  cache.erase(name);
97  }
98  std::shared_ptr<T> v = std::make_shared<E>(row, col);
99  cache.emplace(name, v);
100  return v;
101  }
