31  public:
32  /**
33  * \brief Directivity of T4010A1
34  * \param theta_deg zenith angle in degree
35  * \return directivity
36  */
37  static double t4010a1(double theta_deg) {
38  theta_deg = std::abs(theta_deg);
39  while (theta_deg > 90) theta_deg = std::abs(180 - theta_deg);
40  const auto i = static_cast<size_t>(std::ceil(theta_deg / 10));
41  if (i == 0) return 1;
42  const auto a = DIR_COEFFICIENT_A[i - 1];
43  const auto b = DIR_COEFFICIENT_B[i - 1];
44  const auto c = DIR_COEFFICIENT_C[i - 1];
45  const auto d = DIR_COEFFICIENT_D[i - 1];
46  const auto x = theta_deg - static_cast<double>(i - 1) * 10;
47  return a + b * x + c * x * x + d * x * x * x;
48  }
49 };

51 inline std::complex<double> transfer(const core::Vector3& trans_pos, const core::Vector3& trans_norm, const core::Vector3& target_pos,
52  const double wave_number, const double attenuation) {
53  const auto diff = target_pos - trans_pos;
54  const auto dist = diff.norm();
55  const auto theta = std::atan2(diff.dot(trans_norm), dist * trans_norm.norm()) * 180.0 / M_PI;
56  const auto directivity = Directivity::t4010a1(theta);
57  return directivity / dist * std::exp(std::complex<double>(-dist * attenuation, -wave_number * dist));
58 }
