27 class Holo : public core::Gain {
28  public:
29  Holo(BackendPtr backend, std::vector<core::Vector3> foci, const std::vector<double>& amps) : _backend(std::move(backend)), _foci(std::move(foci)) {
30  if (_foci.size() != amps.size()) throw core::GainBuildError("The size of foci and amps are not the same");
31  _amps.reserve(amps.size());
32  for (const auto amp : amps) _amps.emplace_back(complex(amp, 0.0));
33  }
34  ~Holo() override = default;
35  Holo(const Holo& v) noexcept = default;

44  protected:

231 /**
232  * @brief Gain to produce multiple focal points with Greedy algorithm.
233  * Refer to Shun suzuki, et al. “Radiation Pressure Field Reconstruction for Ultrasound Midair Haptics by Greedy Algorithm with Brute-Force Search,”
234  * in IEEE Transactions on Haptics, doi: 10.1109/TOH.2021.3076489
235  * @details This method is computed on the CPU regardless of the Backend.
236  */
237 class Greedy final : public Holo {
238  public:
239  /**
240  * @brief Generate function
241  * @param[in] foci focal points
242  * @param[in] amps amplitudes of the foci
243  * @param[in] phase_div resolution of the phase to be searched
244  */
245  static std::shared_ptr<Greedy> create(const std::vector<core::Vector3>& foci, const std::vector<double>& amps, const size_t phase_div = 16) {
246  return std::make_shared<Greedy>(foci, amps, phase_div);
247  }
248 
249  void calc(const core::GeometryPtr& geometry) override;
250  Greedy(const std::vector<core::Vector3>& foci, const std::vector<double>& amps, const size_t phase_div) : Holo(nullptr, foci, amps) {
251  this->_phases.reserve(phase_div);
252  for (size_t i = 0; i < phase_div; i++)
253  this->_phases.emplace_back(std::exp(complex(0., 2.0 * M_PI * static_cast<double>(i) / static_cast<double>(phase_div))));
254  }
255 
256  private:
257  std::vector<complex> _phases;
258 };
