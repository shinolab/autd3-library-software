30 class Holo : public core::Gain {
31  public:
32  Holo(BackendPtr backend, std::vector<core::Vector3> foci, const std::vector<double>& amps) : _backend(std::move(backend)), _foci(std::move(foci)) {
33  if (this->_foci.size() != amps.size()) throw core::exception::GainBuildError("The size of foci and amps are not the same");
34  this->_amps.reserve(amps.size());
35  for (const auto amp : amps) this->_amps.emplace_back(complex(amp, 0.0));
36  }
37  ~Holo() override = default;
38  Holo(const Holo& v) noexcept = default;
39  Holo& operator=(const Holo& obj) = default;
40  Holo(Holo&& obj) = default;
41  Holo& operator=(Holo&& obj) = default;
42 
43  std::vector<core::Vector3>& foci() { return this->_foci; }
44  std::vector<complex>& amplitudes() { return this->_amps; }
45 
46  protected:
47  BackendPtr _backend;
48  std::vector<core::Vector3> _foci;
49  std::vector<complex> _amps;
50 };
