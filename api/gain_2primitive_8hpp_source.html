48 class Grouped final : public Gain {
49  public:
50  /**
51  * \brief Decide which device outputs which Gain
52  * \param device_id device id
53  * \param gain gain
54  */
55  template <class T>
56  std::enable_if_t<core::type_traits::is_gain_v<T>> add(const size_t device_id, T&& gain) {
57  Gain& g = core::type_traits::to_gain(gain);
58 
59  g.build(_geometry);
60 
61  if (device_id < _geometry.num_devices())
62  std::memcpy(&this->_tmp_data[device_id * core::NUM_TRANS_IN_UNIT], &g.data()[device_id * core::NUM_TRANS_IN_UNIT],
63  core::NUM_TRANS_IN_UNIT * sizeof(core::Drive));
64  }
65 
66  void calc(const core::Geometry& geometry) override {
67  std::memcpy(this->_data.data(), this->_tmp_data.data(), this->_data.size() * sizeof(core::Drive));
68  }
69 
70  explicit Grouped(const core::Geometry& geometry) : Gain(), _geometry(geometry) { this->_tmp_data.resize(_geometry.num_transducers()); }
71  ~Grouped() override = default;
72  Grouped(const Grouped& v) noexcept = delete;
73  Grouped& operator=(const Grouped& obj) = delete;
74  Grouped(Grouped&& obj) = default;
75  Grouped& operator=(Grouped&& obj) = delete;
76 
77  private:
78  const core::Geometry& _geometry;
79  std::vector<core::Drive> _tmp_data;
80 };
