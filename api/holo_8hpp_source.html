27 class Holo : public core::Gain {
28  public:
29  Holo(BackendPtr backend, std::vector<core::Vector3> foci, const std::vector<double>& amps) : _backend(std::move(backend)), _foci(std::move(foci)) {
30  if (this->_foci.size() != amps.size()) throw core::exception::GainBuildError("The size of foci and amps are not the same");
31  this->_amps.reserve(amps.size());
32  for (const auto amp : amps) this->_amps.emplace_back(complex(amp, 0.0));
33  }
34  ~Holo() override = default;
35  Holo(const Holo& v) noexcept = delete;
36  Holo& operator=(const Holo& obj) = delete;
37  Holo(Holo&& obj) = default;
38  Holo& operator=(Holo&& obj) = default;
39 
40  [[nodiscard]] const std::vector<core::Vector3>& foci() const { return this->_foci; }
41  [[nodiscard]] const std::vector<complex>& amplitudes() const { return this->_amps; }
42 
43  protected:
44  BackendPtr _backend;
45  std::vector<core::Vector3> _foci;
46  std::vector<complex> _amps;
47 };
