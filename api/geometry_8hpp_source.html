54  * \brief Device contains an AUTD device geometry.
55  */
56 struct Device {
57  Device(const size_t id, const Vector3& position, const Quaternion& quaternion)
58  : _id(id),
59  _x_direction(quaternion * Vector3(1, 0, 0)),
60  _y_direction(quaternion * Vector3(0, 1, 0)),
61  _z_direction(quaternion * Vector3(0, 0, 1)) {
62  const Eigen::Transform<double, 3, Eigen::Affine> transform_matrix = Eigen::Translation<double, 3>(position) * quaternion;
63  size_t i = id * NUM_TRANS_IN_UNIT;
64  for (size_t y = 0; y < NUM_TRANS_Y; y++)
65  for (size_t x = 0; x < NUM_TRANS_X; x++) {
66  if (is_missing_transducer(x, y)) continue;
67  const auto local_pos = Vector4(static_cast<double>(x) * TRANS_SPACING_MM, static_cast<double>(y) * TRANS_SPACING_MM, 0, 1);
68  const Vector4 global_pos = transform_matrix * local_pos;
69  _transducers.emplace_back(i++, global_pos[0], global_pos[1], global_pos[2]);
70  }
71  _global_to_local = transform_matrix.inverse();
72  }
73 
74  Device(const size_t id, const Vector3& position, const Vector3& euler_angles)
75  : Device(id, position,
76  Eigen::AngleAxis<double>(euler_angles.x(), Vector3::UnitZ()) * Eigen::AngleAxis<double>(euler_angles.y(), Vector3::UnitY()) *
77  Eigen::AngleAxis<double>(euler_angles.z(), Vector3::UnitZ())) {}

85  * @brief Convert a global position to a local position
86  */
87  [[nodiscard]] Vector3 to_local_position(const Vector3& global_position) const {
88  const auto homo = Vector4(global_position[0], global_position[1], global_position[2], 1);
89  const Vector4 local_position = _global_to_local * homo;
90  return {local_position[0], local_position[1], local_position[2]};
91  }
