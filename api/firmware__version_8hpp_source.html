43  static std::string firmware_version_map(const uint16_t version_number) {
44  std::stringstream ss;
45  if (version_number == 0) {
46  return "older than v0.4";
47  }
48  if (version_number <= 6) {
49  ss << "v0." << version_number + 3;
50  return ss.str();
51  }
52  if (0x000A <= version_number && version_number <= 0x0013) {
53  ss << "v1." << version_number - 0x000A;
54  return ss.str();
55  }
56  if (version_number == 0xFFFF) {
57  return "emulator";
58  }
59  if ((version_number & 0xF000) == 0xF000) {
60  ss << "v0." << version_number - 0xF000 + 1 << "-freq-shift";
61  return ss.str();
62  }
63 
64  ss << "unknown: " << version_number;
65  return ss.str();
66  }
