90  auto* header = reinterpret_cast<GlobalHeader*>(tx.data());
91  header->msg_id = msg_id;
92  constexpr auto fpga_mask = OUTPUT_BALANCE | SILENT | FORCE_FAN;
93  header->fpga_ctrl_flags = (header->fpga_ctrl_flags & ~fpga_mask) | (fpga_ctrl_flag & fpga_mask);
94  header->cpu_ctrl_flags = cpu_ctrl_flag;
95  header->mod_size = 0;
96 
97  tx.num_bodies() = 0;
98 
99  if (is_finished()) return;
100 
101  size_t offset = 0;
102  if (_sent == 0) {
103  header->cpu_ctrl_flags |= MOD_BEGIN;
104  const auto div = static_cast<uint16_t>(sampling_freq_div_ratio() - 1);
105  header->mod[0] = static_cast<uint8_t>(div & 0xFF);
106  header->mod[1] = static_cast<uint8_t>(div >> 8 & 0xFF);
107  offset += 2;
108  }
109  const auto mod_size = static_cast<uint8_t>(std::clamp(_buffer.size() - _sent, size_t{0}, MOD_FRAME_SIZE - offset));
110  if (_sent + mod_size >= _buffer.size()) header->cpu_ctrl_flags |= MOD_END;
111  header->mod_size = mod_size;
112 
113  std::memcpy(&header->mod[offset], &_buffer[_sent], mod_size);
114  _sent += mod_size;
115  }
