102 /**
103  * @brief PointSequence provides a function to display the focus sequentially and periodically.
104  * @details PointSequence uses a timer on the FPGA to ensure that the focus is precisely timed.
105  * PointSequence currently has the following three limitations.
106  * 1. The maximum number of control points is autd::core::POINT_SEQ_BUFFER_SIZE_MAX.
107  * 2. The sampling interval of control points is an integer multiple of 25us and less than or equal to 25us x autd::core::SEQ_SAMPLING_FREQ_DIV_MAX.
108  * 3. Only a single focus can be displayed at a certain moment.
109  */

114  void set(const int32_t x, const int32_t y, const int32_t z, const uint8_t duty) {
115  _buf[0] = x & 0xFFFF; // x 0-15 bit
116  uint16_t tmp = x >> 16 & 0x0001; // x 16th bit
117  tmp |= x >> 30 & 0x0002; // x sign bit
118  tmp |= y << 2 & 0xFFFC; // y 0-13 bit
119  _buf[1] = tmp;
120  tmp = y >> 14 & 0x0007; // y 14-16 bit
121  tmp |= y >> 28 & 0x0008; // y sign bit
122  tmp |= z << 4 & 0xFFF0; // z 0-11 bit
123  _buf[2] = tmp;
124  tmp = z >> 12 & 0x001F; // z 12-16 bit
125  tmp |= z >> 26 & 0x0020; // z sign bit
126  tmp |= duty << 6 & 0x3FC0; // duty
127  _buf[3] = tmp;
128  }
129 
130  private:
131  uint16_t _buf[4];
132  };

184  void pack(const Geometry& geometry, TxDatagram& tx) override {
185  auto* header = tx.header();
186 
187  if (_wait_on_sync) header->cpu_ctrl_flags |= WAIT_ON_SYNC;
188  header->fpga_ctrl_flags |= OUTPUT_ENABLE;
189  header->fpga_ctrl_flags |= SEQ_MODE;
190  header->fpga_ctrl_flags &= ~SEQ_GAIN_MODE;
191 
192  if (is_finished()) return;
193 
194  tx.num_bodies() = geometry.num_devices();
195 
196  size_t offset = 1;
197  header->cpu_ctrl_flags |= WRITE_BODY;
198  if (_sent == 0) {
199  header->cpu_ctrl_flags |= SEQ_BEGIN;
200  for (const auto& device : geometry) {
201  auto* cursor = reinterpret_cast<uint16_t*>(tx.body(device.id()));
202  cursor[1] = static_cast<uint16_t>(_freq_div_ratio - 1);
203  cursor[2] = static_cast<uint16_t>(geometry.wavelength() * 1000);
204  }
205  offset += 4;
206  }
207  const auto send_size = std::clamp(_control_points.size() - _sent, size_t{0}, sizeof(uint16_t) * (NUM_TRANS_IN_UNIT - offset) / sizeof(SeqFocus));
208  if (_sent + send_size >= _control_points.size()) header->cpu_ctrl_flags |= SEQ_END;
209 
210  const auto fixed_num_unit = 256.0 / geometry.wavelength();
211  for (const auto& device : geometry) {
212  auto* cursor = reinterpret_cast<uint16_t*>(tx.body(device.id()));
213  cursor[0] = static_cast<uint16_t>(send_size);
214  auto* focus_cursor = reinterpret_cast<SeqFocus*>(&cursor[offset]);
215  for (size_t i = _sent; i < _sent + send_size; i++, focus_cursor++) {
216  const auto v = (device.to_local_position(_control_points[i]) * fixed_num_unit).cast<int32_t>();
217  focus_cursor->set(v.x(), v.y(), v.z(), _duties[i]);
218  }
219  }
220  _sent += send_size;
221  }
222 
223  [[nodiscard]] bool is_finished() const override { return _sent == _control_points.size(); }

331  void pack(const Geometry& geometry, TxDatagram& tx) override {
332  auto* header = tx.header();
333 
334  if (_wait_on_sync) header->cpu_ctrl_flags |= WAIT_ON_SYNC;
335  header->fpga_ctrl_flags |= OUTPUT_ENABLE;
336  header->fpga_ctrl_flags |= SEQ_MODE;
337  header->fpga_ctrl_flags |= SEQ_GAIN_MODE;
338 
339  if (is_finished()) return;
340 
341  tx.num_bodies() = geometry.num_devices();
342 
343  header->cpu_ctrl_flags |= WRITE_BODY;
344  const auto sent = static_cast<size_t>(_gain_mode);
345  if (_sent == 0) {
346  header->cpu_ctrl_flags |= SEQ_BEGIN;
347  for (const auto& device : geometry) {
348  auto* cursor = reinterpret_cast<uint16_t*>(tx.body(device.id()));
349  cursor[0] = static_cast<uint16_t>(sent);
350  cursor[1] = static_cast<uint16_t>(_freq_div_ratio - 1);
351  cursor[2] = static_cast<uint16_t>(_gain_drives.size());
352  }
353  _sent += 1;
354  return;
355  }
356 
357  if (_sent + sent > _gain_drives.size()) header->cpu_ctrl_flags |= SEQ_END;
358 
359  const auto gain_idx = _sent - 1;
360  switch (_gain_mode) {
361  case GAIN_MODE::DUTY_PHASE_FULL: {
362  auto* cursor = reinterpret_cast<Drive*>(tx.body(0));
363  std::memcpy(cursor, _gain_drives[gain_idx].data(), _gain_drives[gain_idx].size() * sizeof(Drive));
364  } break;
365  case GAIN_MODE::PHASE_FULL:
366  for (const auto& dev : geometry) {
367  auto* cursor = reinterpret_cast<PhaseDrive*>(tx.body(dev.id()));
368  for (const auto& trans : dev)
369  cursor[trans.id()].set(_gain_drives[gain_idx][trans.id()].phase,
370  gain_idx + 1 >= _gain_drives.size() ? 0x00 : _gain_drives[gain_idx + 1][trans.id()].phase);
371  }
372  break;
373  case GAIN_MODE::PHASE_HALF:
374  for (const auto& dev : geometry) {
375  auto* cursor = reinterpret_cast<HalfPhaseDrive*>(tx.body(dev.id()));
376  for (const auto& trans : dev)
377  cursor[trans.id()].set(_gain_drives[gain_idx][trans.id()].phase,
378  gain_idx + 1 >= _gain_drives.size() ? 0x00 : _gain_drives[gain_idx + 1][trans.id()].phase,
379  gain_idx + 2 >= _gain_drives.size() ? 0x00 : _gain_drives[gain_idx + 2][trans.id()].phase,
380  gain_idx + 3 >= _gain_drives.size() ? 0x00 : _gain_drives[gain_idx + 3][trans.id()].phase);
381  }
382  break;
383  }
384  _sent += sent;
385  }
