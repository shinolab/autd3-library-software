41  /**
42  * @brief Set frequency of the sequence
43  * @param[in] freq Frequency of the sequence
44  * @details Sequence mode has some constraints, which determine the actual frequency of the sequence.
45  * @return double Actual frequency of sequence
46  */
47  double set_frequency(const double freq) {
48  const auto sample_freq = static_cast<double>(this->size()) * freq;
49  this->_freq_div_ratio =
50  std::clamp<size_t>(static_cast<size_t>(std::round(static_cast<double>(SEQ_BASE_FREQ) / sample_freq)), 1, SEQ_SAMPLING_FREQ_DIV_MAX);
51  return this->frequency();
52  }

56  */
57  [[nodiscard]] double frequency() const { return this->sampling_freq() / static_cast<double>(this->size()); }

60  * @return period of sequence in micro seconds
61  */
62  [[nodiscard]] size_t period_us() const { return this->sampling_period_us() * this->size(); }

64  /**
65  * The sampling period is limited to an integer multiple of 25us. Therefore, the sampling frequency must be 40kHz/N, where N=1, 2, ...,
66  * autd::core::SEQ_SAMPLING_FREQ_DIV_MAX.
67  * @return double Sampling frequency of sequence
68  */
69  [[nodiscard]] double sampling_freq() const { return static_cast<double>(SEQ_BASE_FREQ) / static_cast<double>(this->_freq_div_ratio); }
70 
71  /**
72  * @return sampling period of sequence in micro seconds
73  */
74  [[nodiscard]] size_t sampling_period_us() const noexcept { return _freq_div_ratio * 1000000 / SEQ_BASE_FREQ; }
75 
76  /**
77  * The sampling frequency division ratio means the autd::core::SEQ_BASE_FREQ/(sampling frequency) = (sampling period)/25us.

87 /**
88  * @brief PointSequence provides a function to display the focus sequentially and periodically.
89  * @details PointSequence uses a timer on the FPGA to ensure that the focus is precisely timed.
90  * PointSequence currently has the following three limitations.
91  * 1. The maximum number of control points is autd::core::POINT_SEQ_BUFFER_SIZE_MAX.
92  * 2. The sampling interval of control points is an integer multiple of 25us and less than or equal to 25us x autd::core::SEQ_SAMPLING_FREQ_DIV_MAX.
93  * 3. Only a single focus can be displayed at a certain moment.
94  */

106  /**
107  * @brief Add control point
108  * @param[in] point control point
109  * @param[in] duty duty ratio
110  */
111  void add_point(const Vector3& point, const uint8_t duty = 0xFF) {
112  if (this->_control_points.size() + 1 > POINT_SEQ_BUFFER_SIZE_MAX)
113  throw exception::SequenceBuildError(
114  std::string("Point sequence buffer overflow. Maximum available buffer size is " + std::to_string(POINT_SEQ_BUFFER_SIZE_MAX)));
115 
116  this->_control_points.emplace_back(point);
117  this->_duties.emplace_back(duty);
118  }
119 
120  /**
121  * @brief Add control points
122  * @param[in] points control points
123  * @param[in] duties duty ratios
124  * @details duties.resize(points.size(), 0xFF) will be called.
125  */
126  void add_points(const std::vector<Vector3>& points, std::vector<uint8_t>& duties) {
127  if (this->_control_points.size() + points.size() > POINT_SEQ_BUFFER_SIZE_MAX)
128  throw exception::SequenceBuildError(
129  std::string("Point sequence buffer overflow. Maximum available buffer size is " + std::to_string(POINT_SEQ_BUFFER_SIZE_MAX)));
130 
131  this->_control_points.reserve(this->_control_points.size() + points.size());
132  this->_control_points.insert(this->_control_points.end(), points.begin(), points.end());
133 
134  duties.resize(points.size(), 0xFF);
135  this->_duties.reserve(this->_duties.size() + duties.size());
136  this->_duties.insert(this->_duties.end(), duties.begin(), duties.end());
137  }
