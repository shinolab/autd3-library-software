209 /**
210  * \brief check if the data which have msg_id have been processed in the devices.
211  * \param msg_id message id
212  * \param rx pointer to received data
213  * \return whether the data have been processed
214  */
215 static bool is_msg_processed(const uint8_t msg_id, const RxDatagram& rx) {
216  size_t processed = 0;
217  for (auto& msg : rx)
218  if (msg.msg_id == msg_id) processed++;
219  return processed == rx.num_messages();
