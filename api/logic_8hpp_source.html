25 static bool is_msg_processed(const size_t num_devices, const uint8_t msg_id, const RxDatagram& rx) {
26  size_t processed = 0;
27  for (auto& [ack, rx_msg_id] : rx)
28  if (rx_msg_id == msg_id) processed++;
29  return processed == num_devices;
30 }
