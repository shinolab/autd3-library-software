34  static uint8_t get_id() {
35  static std::atomic id{MSG_NORMAL_BASE};
36 
37  if (uint8_t expected = 0xff; !id.compare_exchange_weak(expected, MSG_NORMAL_BASE)) id.fetch_add(0x01);
38 
39  return id.load();
40  }

42  /**
43  * \brief check if the data which have msg_id have been processed in the devices.
44  * \param num_devices number of devices
45  * \param msg_id message id
46  * \param rx pointer to received data
47  * \return whether the data have been processed
48  */
49  static bool is_msg_processed(const size_t num_devices, const uint8_t msg_id, const uint8_t* const rx) {
50  size_t processed = 0;
51  for (size_t dev = 0; dev < num_devices; dev++)
52  if (const uint8_t proc_id = rx[dev * 2 + 1]; proc_id == msg_id) processed++;
53  return processed == num_devices;

71  /**
72  * \brief Pack header with modulation data
73  * \param mod Modulation
74  * \param fpga_ctrl_flag fpga control flag
75  * \param cpu_ctrl_flag cpu control flag
76  * \param[out] data pointer to transmission data
77  * \param[out] mod_sent length of modulation data already sent
78  * \return message id
79  */
80  static uint8_t pack_header(const ModulationPtr& mod, const uint8_t fpga_ctrl_flag, const uint8_t cpu_ctrl_flag, uint8_t* const data,
81  size_t* const mod_sent) {
82  const uint8_t msg_id = get_id();
83  pack_header(msg_id, fpga_ctrl_flag, cpu_ctrl_flag, data);
84  if (mod == nullptr || *mod_sent >= mod->buffer().size()) return msg_id;
85 
86  auto* header = reinterpret_cast<GlobalHeader*>(data);
87  size_t offset = 0;
88  if (*mod_sent == 0) {
89  header->cpu_ctrl_flags |= MOD_BEGIN;
90  const auto div = static_cast<uint16_t>(mod->sampling_freq_div_ratio() - 1);
91  header->mod[0] = static_cast<uint8_t>(div & 0xFF);
92  header->mod[1] = static_cast<uint8_t>(div >> 8 & 0xFF);
93  offset += 2;
94  }
95  const auto mod_size = static_cast<uint8_t>(std::clamp(mod->buffer().size() - *mod_sent, size_t{0}, MOD_FRAME_SIZE - offset));
96  if (*mod_sent + mod_size >= mod->buffer().size()) header->cpu_ctrl_flags |= MOD_END;
97  header->mod_size = mod_size;
98 
99  std::memcpy(&header->mod[offset], &mod->buffer()[*mod_sent], mod_size);
100  *mod_sent += mod_size;
101 
102  return msg_id;
103  }

106  * \brief Pack data body which contain phase and duty data of each transducer.

123  /**
124  * \brief Pack data body with sequence data
125  * \param seq Sequence
126  * \param geometry Geometry
127  * \param[out] data pointer to transmission data
128  * \return size_t size to send
129  * \details This function must be called after pack_header
130  */
131  static size_t pack_body(const PointSequencePtr& seq, const Geometry& geometry, uint8_t* data) {
132  if (seq == nullptr || seq->sent() == seq->control_points().size()) return sizeof(GlobalHeader);
133 
134  auto* cursor = reinterpret_cast<uint16_t*>(data + sizeof(GlobalHeader));
135  size_t offset = 1;
136  auto* header = reinterpret_cast<GlobalHeader*>(data);
137  header->cpu_ctrl_flags |= WRITE_BODY;
138  if (seq->sent() == 0) {
139  header->cpu_ctrl_flags |= SEQ_BEGIN;
140  for (const auto& device : geometry) {
141  cursor[device.id() * NUM_TRANS_IN_UNIT + 1] = static_cast<uint16_t>(seq->sampling_freq_div_ratio() - 1);
142  cursor[device.id() * NUM_TRANS_IN_UNIT + 2] = static_cast<uint16_t>(geometry.wavelength() * 1000);
143  }
144  offset += 4;
145  }
146  const auto send_size =
147  std::clamp(seq->control_points().size() - seq->sent(), size_t{0}, sizeof(uint16_t) * (NUM_TRANS_IN_UNIT - offset) / sizeof(SeqFocus));
148  if (seq->sent() + send_size >= seq->control_points().size()) header->cpu_ctrl_flags |= SEQ_END;
149 
150  const auto fixed_num_unit = 256.0 / geometry.wavelength();
151  for (const auto& device : geometry) {
152  cursor[0] = static_cast<uint16_t>(send_size);
153  auto* focus_cursor = reinterpret_cast<SeqFocus*>(&cursor[offset]);
154  for (size_t i = seq->sent(); i < seq->sent() + send_size; i++, focus_cursor++) {
155  const auto v = (device.to_local_position(seq->control_point(i)) * fixed_num_unit).cast<int32_t>();
156  focus_cursor->set(v.x(), v.y(), v.z(), seq->duty(i));
157  }
158  cursor += NUM_TRANS_IN_UNIT;
159  }
160  seq->sent() += send_size;
161  return sizeof(GlobalHeader) + geometry.num_transducers() * sizeof(uint16_t);
162  }

164  /**
165  * \brief Pack data body with sequence data
166  * \param seq Sequence
167  * \param geometry Geometry
168  * \param[out] data pointer to transmission data
169  * \return size_t size to send
170  * \details This function must be called after pack_header
171  */
172  static size_t pack_body(const GainSequencePtr& seq, const Geometry& geometry, uint8_t* data) {
173  if (seq == nullptr || seq->sent() >= seq->gains().size() + 1) return sizeof(GlobalHeader);
174 
175  auto* header = reinterpret_cast<GlobalHeader*>(data);
176  header->cpu_ctrl_flags |= WRITE_BODY;
177  const auto seq_sent = static_cast<size_t>(seq->gain_mode());
178  if (seq->sent() == 0) {
179  header->cpu_ctrl_flags |= SEQ_BEGIN;
180  auto* cursor = reinterpret_cast<uint16_t*>(data + sizeof(GlobalHeader));
181  for (const auto& device : geometry) {
182  cursor[device.id() * NUM_TRANS_IN_UNIT] = static_cast<uint16_t>(seq_sent);
183  cursor[device.id() * NUM_TRANS_IN_UNIT + 1] = static_cast<uint16_t>(seq->sampling_freq_div_ratio() - 1);
184  cursor[device.id() * NUM_TRANS_IN_UNIT + 2] = static_cast<uint16_t>(seq->size());
185  }
186  seq->sent()++;
187  return sizeof(GlobalHeader) + geometry.num_transducers() * sizeof(uint16_t);
188  }
189 
190  if (seq->sent() + seq_sent > seq->gains().size()) header->cpu_ctrl_flags |= SEQ_END;
191 
192  auto* cursor = reinterpret_cast<uint16_t*>(data + sizeof(GlobalHeader));
193  const auto gain_idx = seq->sent() - 1;
194  switch (seq->gain_mode()) {
195  case GAIN_MODE::DUTY_PHASE_FULL:
196  std::memcpy(cursor, seq->gains()[gain_idx]->data().data(), seq->gains()[gain_idx]->data().size() * sizeof(uint16_t));
197  break;
198  case GAIN_MODE::PHASE_FULL:
199  for (const auto& dev : geometry)
200  for (const auto& trans : dev) {
201  cursor[2 * trans.id()] = static_cast<uint8_t>(seq->gains()[gain_idx]->data()[trans.id()].phase);
202  cursor[2 * trans.id() + 1] =
203  static_cast<uint8_t>(gain_idx + 1 >= seq->size() ? 0x00 : seq->gains()[gain_idx + 1]->data()[trans.id()].phase);
204  }
205  break;
206  case GAIN_MODE::PHASE_HALF:
207  for (const auto& dev : geometry)
208  for (const auto& trans : dev) {
209  const auto phase1 = static_cast<uint8_t>(seq->gains()[gain_idx]->data()[trans.id()].phase >> 4 & 0x0F);
210  const auto phase2 =
211  static_cast<uint8_t>(gain_idx + 1 >= seq->size() ? 0x00 : seq->gains()[gain_idx + 1]->data()[trans.id()].phase & 0xF0);
212  const auto phase3 =
213  static_cast<uint8_t>(gain_idx + 2 >= seq->size() ? 0x00 : seq->gains()[gain_idx + 2]->data()[trans.id()].phase >> 4 & 0x0F);
214  const auto phase4 =
215  static_cast<uint8_t>(gain_idx + 3 >= seq->size() ? 0x00 : seq->gains()[gain_idx + 3]->data()[trans.id()].phase & 0xF0);
216  cursor[2 * trans.id()] = utils::pack_to_u16(phase2, phase1);
217  cursor[2 * trans.id() + 1] = utils::pack_to_u16(phase4, phase3);
218  }
219  break;
220  }
221  seq->sent() += seq_sent;
222  return sizeof(GlobalHeader) + geometry.num_transducers() * sizeof(uint16_t);
223  }
