48 /**
49  * \brief Device contains an AUTD device geometry.
50  */
51 struct Device {
52  static Device create(const Vector3& position, const Quaternion& quaternion) {
53  const Eigen::Transform<double, 3, Eigen::Affine> transform_matrix = Eigen::Translation<double, 3>(position) * quaternion;
54  const Vector3 x_direction = quaternion * Vector3(1, 0, 0);
55  const Vector3 y_direction = quaternion * Vector3(0, 1, 0);
56  const Vector3 z_direction = quaternion * Vector3(0, 0, 1);
57 
58  auto global_trans_positions = std::make_unique<Vector3[]>(NUM_TRANS_IN_UNIT);
59 
60  auto index = 0;
61  for (size_t y = 0; y < NUM_TRANS_Y; y++)
62  for (size_t x = 0; x < NUM_TRANS_X; x++) {
63  if (is_missing_transducer(x, y)) continue;
64  const Vector4 local_pos = Vector4(static_cast<double>(x) * TRANS_SPACING_MM, static_cast<double>(y) * TRANS_SPACING_MM, 0, 1);
65  const Vector4 global_pos = transform_matrix * local_pos;
66  global_trans_positions[index++] = Vector3(global_pos[0], global_pos[1], global_pos[2]);
67  }
68 
69  return Device{x_direction, y_direction, z_direction, std::move(global_trans_positions), transform_matrix.inverse()};
70  }
71 
72  static Device create(const Vector3& position, const Vector3& euler_angles) {
73  const Quaternion quaternion = Eigen::AngleAxis<double>(euler_angles.x(), Vector3::UnitZ()) *
74  Eigen::AngleAxis<double>(euler_angles.y(), Vector3::UnitY()) *
75  Eigen::AngleAxis<double>(euler_angles.z(), Vector3::UnitZ());
76  return create(position, quaternion);
77  }
78 
79  Vector3 x_direction;
80  Vector3 y_direction;
81  Vector3 z_direction;
82  std::unique_ptr<Vector3[]> global_trans_positions;
83  Eigen::Transform<double, 3, Eigen::Affine> g2l;
84 };
