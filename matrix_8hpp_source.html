139  Matrix4x4 transform_matrix;

141  transform_matrix(0, 1) = 2 * q.x() * q.y() + 2 * q.y() * q.z();
142  transform_matrix(0, 2) = 2 * q.x() * q.z() - 2 * q.w() * q.y();
143  transform_matrix(0, 3) = v.x();
144  transform_matrix(1, 0) = 2 * q.x() * q.y() - 2 * q.y() * q.z();
145  transform_matrix(1, 1) = 1 - 2 * q.x() * q.x() - 2 * q.z() * q.z();
146  transform_matrix(1, 2) = 2 * q.y() * q.z() + 2 * q.w() * q.x();
147  transform_matrix(1, 3) = v.y();
148  transform_matrix(2, 0) = 2 * q.x() * q.z() + 2 * q.w() * q.y();
149  transform_matrix(2, 1) = 2 * q.y() * q.z() - 2 * q.w() * q.x();
150  transform_matrix(2, 2) = 1 - 2 * q.x() * q.x() - 2 * q.y() * q.y();
151  transform_matrix(2, 3) = v.z();
