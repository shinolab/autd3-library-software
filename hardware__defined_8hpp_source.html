94  _buf[0] = x & 0xFFFF; // x 0-15 bit
95  uint16_t tmp = x >> 16 & 0x0001; // x 16th bit
96  tmp |= x >> 30 & 0x0002; // x sign bit
97  tmp |= y << 2 & 0xFFFC; // y 0-13 bit
98  _buf[1] = tmp;
99  tmp = y >> 14 & 0x0007; // y 14-16 bit
100  tmp |= y >> 28 & 0x0008; // y sign bit
101  tmp |= z << 4 & 0xFFF0; // z 0-11 bit
102  _buf[2] = tmp;
103  tmp = z >> 12 & 0x001F; // z 12-16 bit
104  tmp |= z >> 26 & 0x0020; // z sign bit
105  tmp |= duty << 6 & 0x3FC0; // duty
106  _buf[3] = tmp;
107  }
